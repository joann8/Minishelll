Termcap =
	liste des terminal capabilites
	On peut utiliser une termcap pour effectuer une action
	Permet d'agir sur l'aspect visuelle du prompt

termios =
	structure qui contient les MODES du terminal
	on peut obtenir l'etat du terminal avec :
			tcgetattr
	on peut MODIFIER l'etat du terminal avec :
			tcsetattr
	Permet de desactiver :
		-auto echoing
		-read input in line block and sent by \n | EOF | EOL signal

tgetstr(char *id, char **area)
Permet de recuperer les termcaps sous forme d'echappement
	id = nom du termcap recherche
	area = adresse du buffer utilise lors de "tgetent" (soit NULL);

tputs(const char *str, int affcnt, int (*putc)(int))
	fonction qui execture le termcap recuperer par "tgetstr".
str = termcap recu par tgetstr;
affcnt = nomber de lignes affectees;
putc = pointeur sur fonction
